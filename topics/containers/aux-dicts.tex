\begin{frame}
  \frametitle{Dictionaries}
  \begin{itemize}
    \item Is een \emph{associatieve container}
    \item Associeert \emph{keys} met \emph{values}
    \item Gegeven een key kan de dictionary snel de geassocieerde value opzoeken
  \end{itemize}
  \vskip5mm
  \structure{Voorbeeld}
  \begin{itemize}
    \item Een woordenboek is een associatieve container
    \item Het associeert woorden met hun omschrijving
    \item Key: woord
    \item Value: omschrijving
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Voorbeeld}
  \begin{center}
    \begin{tikzpicture}[key/.style={fill=green!50,minimum width=3cm,minimum height=0.9cm},
                        value/.style={fill=red!50,minimum width=3cm,minimum height=0.9cm}]
                        \draw[fill=blue!50,drop shadow] (-0.5,-0.75) rectangle (7.5,-4.25);
      \foreach[count=\i] \key/\val in {one/un,two/deux,three/trois} {
        \node[key,anchor=north west] (key \i) at (0,-\i) {\ttfamily "\key"};
        \node[value,anchor=north west] (value \i) at (4,-\i) {\ttfamily "\val"};
        \draw[-latex] (key \i) -- (value \i);
      }
    \end{tikzpicture}
  \end{center}

\end{frame}
